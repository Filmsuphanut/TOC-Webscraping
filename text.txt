    function url_domain(data) {
        var a = document.createElement('a');
        a.href = data;
        return getHostName(a.href);
    }

    function getHostName(url) {
        var hostname;
        //find & remove protocol (http, ftp, etc.) and get hostname
        if (url.indexOf("//") > -1) {
            hostname = url.split('/')[2];
        }
        else {
            hostname = url.split('/')[0];
        }
        //find & remove port number
        hostname = hostname.split(':')[0];
        //find & remove "?"
        hostname = hostname.split('?')[0];
        return hostname;
    }

    function CheckCrossDomain(redirectUrl) {
        var currentDomain = window.location.hostname;
        var redirectDomain = url_domain(redirectUrl);
        if (currentDomain == redirectDomain) {
            return false;
        }
        var painList = 'https://ibanking.bangkokbank.com|https://ifunds.bangkokbank.com|https://promptpay.bangkokbank.com|https://getmoney.bangkokbank.com|https://ipay.bangkokbank.com|https://bualuang.bangkokbank.com|https://icareers.bangkokbank.com|https://bizibanking.bangkokbank.com|https://dimenxion.bangkokbank.com|https://icustody.bangkokbank.com|https://itrade.bangkokbank.com|https://ebs.dimenxion.bangkokbank.com|https://developer.bangkokbank.com|https://www.bangkokbank.com/th-TH/Form%20Register/HomeloanRegistration|https://bualuangmbanking.page.link/MutualFund|https://forms.office.com/';
        var exceptionList = painList.split("|");
        var hasException = exceptionList.find(function (itemElement) {
            return url_domain(itemElement) == redirectDomain;
        });
        //Exception Domain false ==show
        if (hasException) {
            return false;
        }
        //Out of Domain
        if (redirectDomain == currentDomain) {
            return false;
        }
        return true;
    }
